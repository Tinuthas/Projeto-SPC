CREATE TABLE t_spc_acesso (
    cd_acesso    NUMBER(7) NOT NULL,
    cd_usuario   NUMBER(7) NOT NULL,
    dt_acesso    DATE NOT NULL
);

ALTER TABLE t_spc_acesso
    ADD CONSTRAINT ck_spc_acesso CHECK (
        (
            cd_pessoa IS NULL
            AND cd_morador IS NOT NULL
            AND cd_vinculo IS NOT NULL
        )
        OR (
            cd_pessoa IS NOT NULL
            AND cd_morador IS NULL
            AND cd_vinculo IS NULL
        )
    );

ALTER TABLE t_spc_acesso ADD CONSTRAINT pk_spc_acesso PRIMARY KEY ( cd_acesso );

CREATE TABLE t_spc_comunicado (
    nr_envio          NUMBER(7) NOT NULL,
    cd_usuario        NUMBER(7) NOT NULL,
    ds_assunto        VARCHAR2(30),

    dt_comunicado     DATE NOT NULL,
    dt_hora_inicio    DATE,
    dt_hora_termino   DATE,

ALTER TABLE t_spc_comunicado ADD CONSTRAINT pk_spc_comunicado PRIMARY KEY ( nr_envio );

CREATE TABLE t_spc_condominio (
    cd_condominio        NUMBER(5) NOT NULL,
    cd_endereco          NUMBER(5) NOT NULL,
    nm_razao_social      VARCHAR2(60) NOT NULL,

        1,

ALTER TABLE t_spc_condominio ADD CONSTRAINT un_spc_condominio UNIQUE ( nm_razao_social );

ALTER TABLE t_spc_encomenda ADD CONSTRAINT pk_spc_encomenda PRIMARY KEY ( cd_morador,
                                                                          cd_encomenda );

CREATE TABLE t_spc_endereco (
    cd_endereco     NUMBER(5) NOT NULL,
    ds_logradouro   VARCHAR2(100) NOT NULL,
    nr_endereco     NUMBER(5) NOT NULL,

    ds_cidade       VARCHAR2(30) NOT NULL,
    ds_bairro       VARCHAR2(30)
);

ALTER TABLE t_spc_endereco ADD CONSTRAINT pk_spc_endereco PRIMARY KEY ( cd_endereco );

CREATE TABLE t_spc_funcionario (
    cd_funcionario   NUMBER(7) NOT NULL,
    dt_admissao      DATE NOT NULL,
    ds_funcao        VARCHAR2(30) NOT NULL

ALTER TABLE t_spc_funcionario ADD CONSTRAINT pk_spc_funcionario PRIMARY KEY ( cd_funcionario );

CREATE TABLE t_spc_morador (

    'Nome da Torre, Opcional';

ALTER TABLE t_spc_morador ADD CONSTRAINT pk_spc_morador PRIMARY KEY ( cd_morador );

CREATE TABLE t_spc_pessoa (
    cd_usuario      NUMBER(7) NOT NULL,
    cd_condominio   NUMBER(5) NOT NULL,
    nr_rg           NUMBER(9) NOT NULL,

    dt_nascimento   DATE NOT NULL,
    ds_sexo         CHAR(1) NOT NULL

ALTER TABLE t_spc_pessoa

        upper(ds_sexo) = 'F'
        OR upper(ds_sexo) = 'M'
        OR upper(ds_sexo) = 'O'
    );

ALTER TABLE t_spc_pessoa ADD CONSTRAINT pk_spc_pessoa PRIMARY KEY ( cd_usuario );

CREATE TABLE t_spc_servico (
    cd_servico        NUMBER(3) NOT NULL,
